#Endpoints

#Reader
POST http://localhost:8080/library/v1/readers/ { name: "John", "surname": "Smith" }
PATCH //localhost:8080/library/v1/readers/1 { "surname" : "Johnson" }

#Book
POST http://localhost:8080/library/v1/books/ { "title": "Lion King", "author": "Justyn Korman", "yearOfRelease": 1994 }
PATCH //localhost:8080/library/v1/books/1 { "yearOfRelease": 1995 }

#BookCopy

PATCH http://localhost:8080/library/v1/book-copies/2/status { "status": "LOST" }
